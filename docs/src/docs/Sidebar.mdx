import { 
  Sidebar, 
  SidebarHeader, 
  SidebarContent, 
  SidebarBrand,
  SidebarMenu,
  SidebarMenuItem,
  SidebarMenuButton,
  SidebarMainContent
} from '@azodik/ui';

function ResponsiveSidebarExample() {
  const [isCollapsed, setIsCollapsed] = useState(false);
  const [isMobile, setIsMobile] = useState(false);

  useEffect(() => {
    const checkScreenSize = () => {
      setIsMobile(window.innerWidth <= 768);
    };
    
    checkScreenSize();
    window.addEventListener('resize', checkScreenSize);
    return () => window.removeEventListener('resize', checkScreenSize);
  }, []);

  return (
    <div style={{ display: 'flex', height: '100vh' }}>
      <Sidebar 
        width={isMobile ? 280 : isCollapsed ? 60 : 250}
        collapsed={!isMobile && isCollapsed}
        className={isMobile ? 'mobile-sidebar' : ''}
      >
        <SidebarHeader>
          <SidebarBrand>Responsive App</SidebarBrand>
        </SidebarHeader>
        <SidebarContent>
          <SidebarMenu>
            <SidebarMenuItem>
              <SidebarMenuButton 
                active
                icon={<HomeIcon size={16} />}
              >
                Home
              </SidebarMenuButton>
            </SidebarMenuItem>

        </SidebarContent>

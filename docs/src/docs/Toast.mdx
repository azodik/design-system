import { Toast, Button, Card, CardContent } from '@azodik/ui';

import { useState } from 'react';

export const BasicToastExample = () => {
  const [showToast, setShowToast] = useState(false);

  return (
    <div className="space-y-4">
      <Button onClick={() => setShowToast(true)}>
        Show Toast
      </Button>
      
      {showToast && (
        <Toast
          variant="info"
          title="Information"
          onClose={() => setShowToast(false)}
          autoClose={3000}
        >
          This is a basic toast notification.
        </Toast>
      )}
    </div>
  );
};

export const ToastVariantsExample = () => {
  const [toasts, setToasts] = useState([]);

  const addToast = (variant, title, message) => {
    const id = Date.now();
    setToasts(prev => [...prev, { id, variant, title, message }]);
  };

  const removeToast = (id) => {
    setToasts(prev => prev.filter(toast => toast.id !== id));
  };

  return (
    <div className="space-y-4">
      <div className="flex gap-2 flex-wrap">
        <Button onClick={() => addToast('success', 'Success!', 'Operation completed successfully.')}>
          Success Toast
        </Button>
        <Button onClick={() => addToast('error', 'Error!', 'Something went wrong.')}>
          Error Toast
        </Button>
        <Button onClick={() => addToast('warning', 'Warning!', 'Please check your input.')}>
          Warning Toast
        </Button>
        <Button onClick={() => addToast('info', 'Info', 'Here is some information.')}>
          Info Toast
        </Button>
      </div>
      
      {toasts.map(toast => (
        <Toast
          key={toast.id}
          variant={toast.variant}
          title={toast.title}
          onClose={() => removeToast(toast.id)}
          autoClose={4000}
        >
          {toast.message}
        </Toast>
      ))}
    </div>
  );
};

# Toast Component

The Toast component provides temporary notifications that appear and disappear automatically. It's perfect for showing success messages, errors, or other brief feedback to users.

## Introduction

The Toast component includes several complementary features to handle various notification and feedback scenarios:

- Toast: the main wrapper for creating temporary notifications that provide feedback without blocking user workflow.
- Toast Variants: multiple notification types (success, warning, error, info) with appropriate icons and styling for different message types.
- Toast Positioning: flexible positioning system with multiple placement options (top-center, top-right, bottom-right) for optimal visibility.
- Toast Timing: automatic dismissal with configurable duration and manual close functionality for different notification needs.
- Toast Customization: flexible styling, icons, titles, and content options for brand-specific notification designs.
- Toast Accessibility: proper ARIA attributes, screen reader announcements, and keyboard navigation support for notification consumption.

### When to use
- Show success messages after actions
- Display error notifications
- Provide brief status updates
- Show confirmation messages
- Display system notifications

## Accessibility

- Toasts are announced to screen readers
- Focus management is handled correctly
- Keyboard navigation is supported
- Color is not the only way to convey information
- Proper ARIA attributes are applied

## Design Guidelines

- Keep toast messages concise and clear
- Use appropriate variants for different message types
- Don't show too many toasts at once
- Provide clear close actions
- Consider mobile positioning and sizing

export const ControlledRadioExample = () => {
  const [selectedLanguage, setSelectedLanguage] = useState('');

  const handleChange = (e) => {
    setSelectedLanguage(e.target.value);
  };

  return (
    <div className="space-y-2">
      <h3>Choose your preferred language:</h3>
      <Radio 
        id="english" 
        name="language" 
        value="english"
        label="English"
        checked={selectedLanguage === 'english'}
        onChange={handleChange}
      />
      <Radio 
        id="spanish" 
        name="language" 
        value="spanish"
        label="Spanish"
        checked={selectedLanguage === 'spanish'}
        onChange={handleChange}

        id="french"

      />
      <p className="text-sm text-gray-500">

export const ValidatedRadioExample = () => {
  const [selectedOption, setSelectedOption] = useState('');
  const [error, setError] = useState('');

  const handleChange = (e) => {
    setSelectedOption(e.target.value);
    if (e.target.value) {
      setError('');
    }
  };

  const handleBlur = () => {
    if (!selectedOption) {
      setError('Please select an option');
    }
  };

  return (
    <div className="space-y-2">
      <h3>Choose your payment method: *</h3>
      <div className="flex items-center space-x-2">
        <Radio 
          id="credit" 
          name="payment" 
          value="credit"
          checked={selectedOption === 'credit'}
          onChange={handleChange}
          onBlur={handleBlur}
        />
        <label htmlFor="credit">Credit Card</label>
      </div>
      <div className="flex items-center space-x-2">
        <Radio 
          id="debit" 
          name="payment" 
          value="debit"
          checked={selectedOption === 'debit'}
          onChange={handleChange}
          onBlur={handleBlur}
        />
        <label htmlFor="debit">Debit Card</label>
      </div>
      <div className="flex items-center space-x-2">
        <Radio 
          id="paypal" 
          name="payment" 
          value="paypal"
          checked={selectedOption === 'paypal'}
          onChange={handleChange}
          onBlur={handleBlur}
        />
        <label htmlFor="paypal">PayPal</label>
      </div>
      {error && <p className="text-sm text-red-500">{error}</p>}
    </div>
  );
};

# Radio Component

The Radio component provides single-choice controls for selecting one option from a group. It's perfect for forms that require exclusive selection.

## Overview

Radio buttons allow users to select exactly one option from a set of mutually exclusive choices. They provide clear visual feedback and ensure data consistency in forms.

### When to use
- Single selection from multiple options
- Choosing one option from a group
- Settings with mutually exclusive choices
- Form fields with limited, exclusive options
- Survey questions with single answers

## Accessibility

- Radio buttons have proper labels and associations
- Screen readers can identify radio button states
- Keyboard navigation is supported
- Focus management is handled correctly
- Proper ARIA attributes are applied

## Design Guidelines

- Always provide clear labels
- Use consistent sizing across your application
- Group related radio buttons together
- Provide clear visual feedback for selection
- Ensure sufficient touch targets on mobile
